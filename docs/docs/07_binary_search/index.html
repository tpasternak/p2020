    from time import time

tablica = [3, 5, 10, 11, 19, 25, 31, 35, 36]
tablica = list(range(1,1000003,3))
szukana = 10000000000000000

t1 = time()
# ALGORYTM 1
for i in range(len(tablica)):
    if tablica[i] == szukana:
        print("Jest")
        break
    elif i + 1 == len(tablica):
        print("Nie ma")
t2 = time()
print(f"Algorytm wykonywal sie {t2-t1:f}")

# ALGORYTM 2
t3 = time()
poczatek = 0
koniec =len(tablica)

while poczatek != koniec:
    srodek = (poczatek + koniec) // 2
    if tablica[srodek] == szukana:
        print("Jest")
        break
    elif srodek == poczatek:
        print("Nie ma")
        break
    elif tablica[srodek] < szukana:
        poczatek = srodek
    else:
        koniec = srodek
t4 = time()
print(f"Algorytm 2 wykonywal sie {t4 - t3:f}")
