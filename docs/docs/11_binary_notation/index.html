def z_formy_dziesietnej(n):
    potega = 1
    while n > potega:
        potega = potega * 2
    potega = potega // 2
    
    lista = ''
    suma = 0
    while potega != 0:
        if suma + potega <= n:
            lista = lista + '1'
            suma += potega
        else:
            lista = lista + '0'
        potega = potega // 2
    return "".join(lista)
    
print(z_formy_dziesietnej(19))
