    Szyfr Cezara

    chr, ord # 

    text = input()
klucz = -3

# ascii
# ord, chr

zaszyfrowany_text = ""
for i in range(len(text)):
    litera = ord(text[i])
    numer_litery = litera % 97
    nowy_numer_litery = (numer_litery + klucz) % 26
    nowy_kod_ascii_litery = nowy_numer_litery + 97
    nowa_litera = chr(nowy_kod_ascii_litery)
    zaszyfrowany_text = zaszyfrowany_text + nowa_litera

print(zaszyfrowany_text)
